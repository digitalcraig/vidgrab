#ytdl-new
/share/Container/youtube-dl/youtube-dl \
    --force-ipv4 \
    --dateafter now-1week \
    --playlist-reverse \
    --write-description \
    --write-thumbnail \
    --write-sub \
    --embed-subs \
    --all-subs \
    --write-annotations \
    --sleep-interval 20 \
    --max-sleep-interval 60 \
    --write-info-json \
    --ignore-errors \
    --no-continue \
    --no-overwrites \
    --no-post-overwrites \
    --embed-thumbnail \
    --add-metadata \
    --download-archive "/share/Multimedia/Internet Videos/archive.txt" \
    --ignore-errors \
    -o "/share/Multimedia/Internet Videos/%(uploader)s/%(uploader)s - %(upload_date)s - %(title)s.%(ext)s" 
    -f bestvideo[ext=mp4]+bestaudio[ext=m4a] 
    --merge-output-format mp4 
    --ffmpeg-location /usr/bin/ffmpeg 
    --batch-file="/share/Multimedia/Internet Videos/youtube.txt" 
    --cookies "/share/Multimedia/Internet Videos/youtube.com_cookies.txt" 
    --cache-dir "/share/Multimedia/Internet Videos/.cache"
